    <script>
        // === CONFIGURAÇÃO CRÍTICA ===
        // COLE AQUI A URL DO SEU GOOGLE APPS SCRIPT (Parte 1)
        const SCRIPT_URL = "https://script.google.com/macros/s/AKfycbwRYIlK5DQP84S_TjzPbgVkLD2isubr-Kh5nWfReWXYMEVJulod3zjfy_O4ABHdSIkX/exec"; 

        // === Lógica do Aplicativo ===
        
        // Exibir Notificações
        function mostrarToast(mensagem, cor = 'bg-slate-800') {
            const toast = document.getElementById('toast');
            toast.textContent = mensagem;
            toast.className = `fixed top-20 left-1/2 transform -translate-x-1/2 z-50 ${cor} text-white px-4 py-3 rounded-lg shadow-xl text-sm font-medium transition-opacity duration-300 w-11/12 max-w-sm text-center`;
            toast.classList.remove('hidden');
            setTimeout(() => toast.classList.add('hidden'), 3000);
        }

        // Navegação de Abas
        function mudarAba(abaId) {
            document.querySelectorAll('.tab-content').forEach(el => el.classList.remove('active'));
            document.getElementById('view-' + abaId).classList.add('active');
            
            document.getElementById('tab-form').classList.replace(abaId === 'form' ? 'text-slate-400' : 'text-emerald-600', abaId === 'form' ? 'text-emerald-600' : 'text-slate-400');
            document.getElementById('tab-sync').classList.replace(abaId === 'sync' ? 'text-slate-400' : 'text-emerald-600', abaId === 'sync' ? 'text-emerald-600' : 'text-slate-400');
            
            if(abaId === 'sync') atualizarListaFila();
        }

        // Capturar GPS
        function capturarGPS() {
            if (!navigator.geolocation) {
                mostrarToast('Seu celular não suporta GPS pelo navegador.', 'bg-red-600');
                return;
            }
            mostrarToast('Buscando satélites...', 'bg-blue-600');
            navigator.geolocation.getCurrentPosition(
                (pos) => {
                    const lat = pos.coords.latitude.toFixed(6);
                    const lng = pos.coords.longitude.toFixed(6);
                    document.getElementById('gpsCoords').value = `${lat}, ${lng}`;
                    mostrarToast('GPS cravado!', 'bg-emerald-600');
                },
                (err) => mostrarToast('Erro no GPS. Ative a localização.', 'bg-red-600'),
                { enableHighAccuracy: true, timeout: 10000 }
            );
        }

        // Processar Foto (Reduzir tamanho para não travar o celular offline)
        function processarFoto(event) {
            const file = event.target.files[0];
            if (!file) return;

            const reader = new FileReader();
            reader.onload = function(e) {
                const img = new Image();
                img.onload = function() {
                    const canvas = document.createElement('canvas');
                    const ctx = canvas.getContext('2d');
                    // Reduz para máx 800px para economizar memória (Storage Limit)
                    const MAX_WIDTH = 800;
                    const scaleSize = MAX_WIDTH / img.width;
                    canvas.width = MAX_WIDTH;
                    canvas.height = img.height * scaleSize;
                    
                    ctx.drawImage(img, 0, 0, canvas.width, canvas.height);
                    const base64 = canvas.toDataURL('image/jpeg', 0.6); // Compressão
                    
                    document.getElementById('fotoBase64').value = base64;
                    document.getElementById('fotoPreview').src = base64;
                    document.getElementById('fotoPreview').classList.remove('hidden');
                    document.getElementById('fotoPlaceholder').classList.add('hidden');
                    document.getElementById('fotoContainer').classList.replace('border-slate-300', 'border-emerald-500');
                };
                img.src = e.target.result;
            };
            reader.readAsDataURL(file);
        }

        // Salvar no Banco de Dados Local (Offline)
        document.getElementById('ecoForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const novoRegistro = {
                ID: 'ECO-' + Date.now().toString(36).toUpperCase(),
                Data_Hora: new Date().toLocaleString('pt-BR'),
                Latitude_Longitude: document.getElementById('gpsCoords').value,
                Tipo_de_Achado: document.getElementById('tipoAchado').value,
                Descricao_Cenario: document.getElementById('descricao').value,
                Foto_Evidencia: document.getElementById('fotoBase64').value
            };

            // Salva na memória do navegador (LocalStorage)
            let registros = JSON.parse(localStorage.getItem('ecoFogoDb')) || [];
            registros.push(novoRegistro);
            localStorage.setItem('ecoFogoDb', JSON.stringify(registros));

            mostrarToast('Salvo em segurança no celular!', 'bg-emerald-600');
            
            // Limpar form
            this.reset();
            document.getElementById('fotoBase64').value = '';
            document.getElementById('fotoPreview').classList.add('hidden');
            document.getElementById('fotoPlaceholder').classList.remove('hidden');
            document.getElementById('fotoContainer').classList.replace('border-emerald-500', 'border-slate-300');
            
            atualizarContadores();
        });

        // Atualizar Tela de Fila
        function atualizarListaFila() {
            const lista = document.getElementById('listaRegistros');
            const registros = JSON.parse(localStorage.getItem('ecoFogoDb')) || [];
            
            if (registros.length === 0) {
                lista.innerHTML = `<div class="bg-white p-8 rounded-xl border border-dashed border-slate-300 text-center text-slate-500">Nenhum dado pendente na memória.</div>`;
                return;
            }

            lista.innerHTML = registros.map((reg, index) => `
                <div class="bg-white p-4 rounded-xl border border-l-4 border-l-amber-500 border-slate-200 shadow-sm">
                    <div class="flex justify-between items-start mb-2">
                        <span class="bg-slate-100 text-slate-700 text-xs font-bold px-2 py-1 rounded">${reg.Tipo_de_Achado}</span>
                        <span class="text-[10px] text-slate-400 font-mono">${reg.ID}</span>
                    </div>
                    <p class="text-xs text-slate-600 mb-1 flex items-center gap-1 font-mono">📍 ${reg.Latitude_Longitude}</p>
                    <p class="text-sm text-slate-800 line-clamp-2">${reg.Descricao_Cenario || "Sem notas."}</p>
                    <div class="mt-3 pt-2 border-t border-slate-100 flex justify-between items-center">
                        <span class="text-[10px] text-slate-400">${reg.Data_Hora}</span>
                        <button onclick="apagarRegistroLocal(${index})" class="text-red-500 text-xs font-semibold hover:underline">Descartar</button>
                    </div>
                </div>
            `).join('');
        }

        function apagarRegistroLocal(index) {
            if(confirm("Apagar esta evidência do celular? Isso não pode ser desfeito.")) {
                let registros = JSON.parse(localStorage.getItem('ecoFogoDb')) || [];
                registros.splice(index, 1);
                localStorage.setItem('ecoFogoDb', JSON.stringify(registros));
                atualizarListaFila();
                atualizarContadores();
            }
        }

        function atualizarContadores() {
            const qtd = (JSON.parse(localStorage.getItem('ecoFogoDb')) || []).length;
            document.getElementById('contadorFila').innerText = qtd;
            const badge = document.getElementById('badgeFila');
            if (qtd > 0) {
                badge.innerText = qtd;
                badge.classList.remove('hidden');
            } else {
                badge.classList.add('hidden');
            }
        }

        // --- MÁGICA DA SINCRONIZAÇÃO (Envio para a Planilha) ---
        async function sincronizarDados() {
            const registros = JSON.parse(localStorage.getItem('ecoFogoDb')) || [];
            
            if (registros.length === 0) {
                mostrarToast('Não há dados para enviar.', 'bg-slate-600');
                return;
            }

            if (!navigator.onLine) {
                mostrarToast('Você está sem internet no momento.', 'bg-red-600');
                return;
            }

            if (SCRIPT_URL === "https://script.google.com/macros/s/AKfycbwRYIlK5DQP84S_TjzPbgVkLD2isubr-Kh5nWfReWXYMEVJulod3zjfy_O4ABHdSIkX/exec") {
                mostrarToast('Configure a SCRIPT_URL no código fonte!', 'bg-red-600');
                return;
            }

            const btn = document.getElementById('btnSync');
            btn.innerHTML = `<svg class="animate-spin -ml-1 mr-2 h-4 w-4 text-white" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24"><circle class="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" stroke-width="4"></circle><path class="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path></svg> Enviando...`;
            btn.disabled = true;

            let enviadosComSucesso = 0;

            for (let i = 0; i < registros.length; i++) {
                const dado = registros[i];
                try {
                    // Prepara os dados para enviar como um formulário (POST)
                    const formData = new URLSearchParams();
                    for (const key in dado) {
                        formData.append(key, dado[key]);
                    }

                    // Envia para o Google Apps Script
                    const resposta = await fetch(SCRIPT_URL, {
                        method: 'POST',
                        body: formData,
                        mode: 'no-cors' // Necessário para evitar bloqueios de segurança do Google
                    });
                    
                    enviadosComSucesso++;
                } catch (erro) {
                    console.error("Falha ao enviar:", erro);
                }
            }

            // Remove da memória apenas os que foram enviados
            registros.splice(0, enviadosComSucesso);
            localStorage.setItem('ecoFogoDb', JSON.stringify(registros));
            
            atualizarListaFila();
            atualizarContadores();
            
            btn.innerHTML = `<svg xmlns="http://www.w3.org/2000/svg" width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M21 15v4a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2v-4"></path><polyline points="17 8 12 3 7 8"></polyline><line x1="12" y1="3" x2="12" y2="15"></line></svg> Enviar Tudo`;
            btn.disabled = false;

            if (enviadosComSucesso > 0) {
                mostrarToast(`${enviadosComSucesso} achado(s) salvo(s) na Planilha!`, 'bg-emerald-600');
            } else {
                mostrarToast('Falha na rede. Tente de novo.', 'bg-red-600');
            }
        }

        // Inicialização
        atualizarContadores();
        window.addEventListener('online', () => {
            document.getElementById('network-indicator').classList.replace('bg-red-500', 'bg-emerald-400');
            document.getElementById('network-text').innerText = 'Online';
        });
        window.addEventListener('offline', () => {
            document.getElementById('network-indicator').classList.replace('bg-emerald-400', 'bg-red-500');
            document.getElementById('network-text').innerText = 'Modo Área Fria';
        });
